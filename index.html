Python Turtle Graphics Code


import turtle
import math

screen = turtle.Screen()
screen.bgcolor("white")
screen.setup(width=1000, height=600)

t = turtle.Turtle()
t.hideturtle()
t.speed(0)
turtle.tracer(False)

flag_width = 900
flag_height = 450
stripe_height = flag_height / 9
wave_amplitude = 10
wave_length = 40

BLUE = "#0D5EAF"

start_x = -flag_width / 2
start_y = flag_height / 2

def draw_wavy_stripe(y, color):
    t.penup()
    t.goto(start_x, y)
    t.setheading(0)
    t.color(color)
    t.begin_fill()

    for x in range(int(flag_width)):
        wave = math.sin(x / wave_length) * wave_amplitude
        t.goto(start_x + x, y + wave)

    for x in range(int(flag_width), -1, -1):
        wave = math.sin(x / wave_length) * wave_amplitude
        t.goto(start_x + x, y - stripe_height + wave)

    t.end_fill()

for i in range(9):
    color = BLUE if i % 2 == 0 else "white"
    draw_wavy_stripe(start_y - i * stripe_height, color)

canton_size = stripe_height * 5
t.penup()
t.goto(start_x, start_y)
t.color(BLUE)
t.begin_fill()
for _ in range(4):
    t.forward(canton_size)
    t.right(90)
t.end_fill()

t.color("white")

t.penup()
t.goto(start_x + canton_size / 2 - stripe_height / 2, start_y)
t.begin_fill()
for _ in range(2):
    t.forward(stripe_height)
    t.right(90)
    t.forward(canton_size)
    t.right(90)
t.end_fill()

t.penup()
t.goto(start_x, start_y - canton_size / 2 + stripe_height / 2)
t.begin_fill()
for _ in range(2):
    t.forward(canton_size)
    t.right(90)
    t.forward(stripe_height)
    t.right(90)
t.end_fill()

turtle.tracer(True)
turtle.done()
